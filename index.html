Korah.js

A lightweight and fresh take on browser-side declarative DOM construction and templating, with zero string parsing.

Visit the Github repo
  • Minifies to (only!) ~0.8kB.
  • Expressive and declarative templates.
  • Extensible tags system.
Vanilla JS Korah.js 
var template = function(data) {
  var div = document.createElement('div'),
      h1 = document.createElement('h1'),
      ul = document.createElement('ul');

  h1.textContent = data.title;

  data.items.forEach(function(text) {
    var li = document.createElement('li');
    li.textContent = text;
    ul.appendChild(li);
  });

  div.appendChild(h1);
  div.appendChild(ul);
  return div;
};
var template = Korah(function(t, data) {
  return t.div(
    t.h1(data.title),
    t.ul(data.items.map(function(text) {
      return t.li(text);
    }))
  );
});
